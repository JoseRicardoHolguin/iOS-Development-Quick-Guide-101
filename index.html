OOP in Swift

Learn classes, inheritance, and protocols to build reusable domain models and app services.

protocol Vehicle {
  var model: String { get }
  func drive()
}

class Car: Vehicle {
  let model: String
  init(model: String) { self.model = model }
  func drive() {
    print("Driving \(model) smoothly")
  }
}

let car = Car(model: "SwiftGT")
car.drive()

Real Use Case

Model app entities, define service protocols, and keep business logic testable and separated.

04

MVVM + @Observable

Use observable view models to keep your SwiftUI UI in sync with state and business data.

class TaskViewModel: ObservableObject {
  @Published var tasks: [String] = []

  func add(_ task: String) {
    tasks.append(task)
  }
}

struct TaskListView: View {
  @StateObject var viewModel = TaskViewModel()

  var body: some View {
    List(viewModel.tasks, id: \.
    self) { task in
      Text(task)
    }
  }
}

Real Use Case

Keep screens reactive and decoupled from network or storage logic in modern Swift apps.

08

REST APIs and JSON

Fetch remote data, decode JSON into Swift models, and display network-driven content.

struct Article: Codable {
  let id: Int
  let title: String
}

let url = URL(string: "https://api.example.com/articles")!
URLSession.shared.dataTask(with: url) { data, _, _ in
  guard let data = data else { return }
  let articles = try? JSONDecoder().decode([Article].self, from: data)
}
.resume()

Real Use Case

Power dashboards, feeds, and API-backed views using Codable data models and network tasks.

09

MapKit

Integrate maps, annotations, and user location to build immersive location-aware experiences.

import MapKit

struct MapView: UIViewRepresentable {
  func makeUIView(context: Context) -> MKMapView {
    MKMapView(frame: .zero)
  }
  func updateUIView(_ view: MKMapView, context: Context) {
    let region = MKCoordinateRegion(center: CLLocationCoordinate2D(latitude: 37.7749, longitude: -122.4194), span: MKCoordinateSpan(latitudeDelta: 0.1, longitudeDelta: 0.1))
    view.setRegion(region, animated: true)
  }
}

Real Use Case

Add maps for event locations, tracking features, delivery apps, and location-aware utilities.

import AVFoundation

func setupCaptureSession() {
  let session = AVCaptureSession()
  guard let camera = AVCaptureDevice.default(for: .video) else { return }
  let input = try? AVCaptureDeviceInput(device: camera)
  if let input { session.addInput(input) }
}
